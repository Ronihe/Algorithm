/**
 *
 * @param {Array} board - the 2D array for the rabbit's garden
 *@returns {Object} Object with row, col, and value keys, the values are Int
 */

function _findStartingSpot(board) {
  let row =
    board.length % 2 === 0
      ? board.length / 2 - 1
      : Math.floor(board.length / 2);

  let col =
    board[0].length % 2 === 0
      ? board[0].length / 2 - 1
      : Math.floor(board.length / 2);
  let val = board[row][col];
  return { row, col, val };
}

/**
 *
 * @param {Array} board - the 2D array for the rabbit's garden
 * @param {Object} currSpot - the current positon of the rabbit with row, col and value of the board
 *@returns {Object} Object with row, col, and value keys, the values are Int
 */

function _findNextSpot(board, currSpot) {
  //check the four directions
  let left = {
    row: currSpot.row - 1,
    col: currSpot.col,
    value: board[currSpot.row - 1][currSpot.col] || 0
  };
  let right = {
    row: currSpot.row + 1,
    col: currSpot.col,
    value: board[currSpot.row + 1][currSpot.col] || 0
  };
  let top = {
    row: currSpot.row,
    col: currSpot.col + 1,
    value: board[currSpot.row][currSpot.col + 1] || 0
  };
  let bottom = {
    row: currSpot.row,
    col: currSpot.col - 1,
    value: board[currSpot.row][currSpot.col - 1] || 0
  };
  let next = Object.values({ left, top, right, bottom }).reduce(
    (accu, currItem) => {
      return accu.value > currItem.value ? accu : currItem;
    }
  );
  return next;
}
